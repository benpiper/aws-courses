; Allocating an elastic IP address
aws ec2 allocate-address
aws ec2 release-address --allocation-id 
aws ec2 describe-addresses

; Launching an instance into a public subnet
aws ec2 describe-subnets --filters Name=cidr-block,Values="10.0.11.0/24"
aws ec2 run-instances --image-id ami-01d025118d8e760db --subnet-id  --instance-type t3.micro --key-name ccnetkeypair
aws ec2 allocate-address
aws ec2 associate-address --instance-id  --allocation-id 
aws ec2 describe-addresses --allocation-ids 
aws ec2 terminate-instances --instance-ids 
aws ec2 release-address --allocation-id 

; Deleting the NAT gateway
aws ec2 describe-nat-gateways
aws ec2 delete-nat-gateway --nat-gateway-id 
aws ec2 release-address --allocation-id 

; Direct Connect and Transit Gateway
aws ec2 create-transit-gateway
aws ec2 create-vpc --cidr-block 172.27.0.0/16
aws ec2 create-subnet --vpc-id  --cidr-block 172.27.1.0/24 --availability-zone us-east-1a
aws ec2 create-transit-gateway-vpc-attachment --transit-gateway-id  --vpc-id  --subnet-ids 
aws ec2 create-vpc --cidr-block 172.28.0.0/16
aws ec2 create-subnet --vpc-id  --cidr-block 172.28.1.0/24 --availability-zone us-east-1a
aws ec2 create-transit-gateway-vpc-attachment --transit-gateway-id  --vpc-id  --subnet-ids 
aws ec2 search-transit-gateway-routes --transit-gateway-route-table-id tgw-rtb-01e158d45848e8522 --filters "Name=type,Values=static,propagated"
aws ec2 describe-transit-gateways
aws ec2 search-transit-gateway-routes --transit-gateway-route-table-id tgw-rtb-0ef0345df00016c92 --filters "Name=type,Values=static,propagated"
